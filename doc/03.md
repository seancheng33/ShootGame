### 第一步：完善子弹脚本 – 增加伤害和来源标记

打开 `bullet.tscn` 对应的脚本 `bullet.gd`，修改为：

```gdscript
extends Area2D

@export var speed := 400.0
var direction := Vector2.UP
@export var damage := 1
var is_player_bullet := true   # true=玩家子弹，false=敌机子弹

func _ready():
    # 根据子弹类型添加到对应分组，便于炸弹清理
    if is_player_bullet:
        add_to_group("player_bullets")
    else:
        add_to_group("enemy_bullets")
    
    # 连接碰撞信号
    area_entered.connect(_on_area_entered)

func _physics_process(delta):
    position += direction * speed * delta
    # 超出屏幕销毁
    var screen = get_viewport_rect()
    if position.y < -100 or position.y > screen.size.y + 100 or \
       position.x < -100 or position.x > screen.size.x + 100:
        queue_free()

func _on_area_entered(other_area: Area2D):
    # 只处理与角色的碰撞（玩家或敌机）
    if is_player_bullet and other_area.get_parent() is CharacterBody2D and other_area.get_parent().has_method("take_damage"):
        var enemy = other_area.get_parent()
        enemy.take_damage(damage)
        queue_free()  # 子弹击中后消失
    elif not is_player_bullet and other_area.get_parent().has_method("take_damage") and other_area.get_parent().name == "Player":
        var player = other_area.get_parent()
        player.take_damage(damage)
        queue_free()
```

> 注意：这里假设玩家和敌机的根节点都是 `CharacterBody2D`，并且它们都有一个 `CollisionShape2D` 子节点用于碰撞检测。同时，玩家和敌机都需要实现 `take_damage(amount)` 方法。

### 第三步：制作普通敌机脚本（支持生命值、分数、死亡特效）

打开 `enemy.tscn` 的脚本 `enemy.gd`，重写为：

```gdscript
extends CharacterBody2D

@export var speed := 150.0
@export var health := 1
@export var score_value := 100   # 普通飞机100分
@export var can_shoot := false    # 普通敌机默认不射击
@export var shoot_cooldown := 1.0

var shoot_timer := 0.0
var bullet_scene : PackedScene

func _ready():
    add_to_group("enemies")
    bullet_scene = preload("res://scenes/bullet.tscn")
    # 如果敌机可以射击，启动计时器
    if can_shoot:
        shoot_timer = randf_range(0.5, 1.5)

func _physics_process(delta):
    velocity = Vector2(0, speed)
    move_and_slide()
    
    # 超出屏幕底部销毁
    if position.y > get_viewport_rect().size.y + 100:
        queue_free()
    
    # 射击逻辑
    if can_shoot:
        if shoot_timer > 0:
            shoot_timer -= delta
        else:
            shoot()
            shoot_timer = shoot_cooldown

func shoot():
    if bullet_scene == null:
        return
    var bullet = bullet_scene.instantiate()
    bullet.position = global_position
    bullet.direction = Vector2.DOWN
    bullet.speed = 300
    bullet.is_player_bullet = false
    bullet.damage = 1
    get_tree().current_scene.add_child(bullet)

func take_damage(amount: int):
    health -= amount
    if health <= 0:
        destroy()

func destroy():
    # 增加分数（需要获取全局分数管理器，稍后实现）
    # 暂时通过信号或直接调用全局单例，这里先打印
    print("Enemy destroyed, score +", score_value)
    # TODO: 增加分数
    # 播放爆炸特效（简单用一个临时 Sprite2D 或粒子）
    # 最后销毁自身
    queue_free()
```

### 第五步：制作 Boss 基类（第八关使用）

Boss 体积较大，生命值高，有多个攻击模式。先创建一个基础 Boss 场景 `boss.tscn`，后续关卡再具体定制。

- 新建场景，根节点 `CharacterBody2D`，命名为 `Boss`。
- 添加 `Sprite2D`，使用较大的 Boss 图片（Kenney 素材中没有专门 Boss，可以用大号敌机图片或从其它素材获取；暂时用 `enemyShip3.png` 放大比例）。
- 添加 `CollisionShape2D`，形状用矩形或圆形，覆盖 Boss 身体。
- 添加脚本 `boss.gd`：

```gdscript
extends CharacterBody2D

@export var health := 20
@export var score_value := 0   # 分数由关卡乘以2000动态计算，稍后设置
@export var speed := 50.0

var bullet_scene : PackedScene
var shoot_pattern := 0
var shoot_timer := 0.0

func _ready():
    add_to_group("enemies")
    bullet_scene = preload("res://scenes/bullet.tscn")
    # 分数在外界设置，例如 set_score(level * 2000)

func _physics_process(delta):
    velocity = Vector2(0, speed)
    move_and_slide()
    
    # Boss 不会超出底部自动销毁，需要击败才消失
    
    # 简单射击模式：每1秒向玩家方向发射子弹
    shoot_timer -= delta
    if shoot_timer <= 0:
        shoot()
        shoot_timer = 1.0

func shoot():
    if bullet_scene == null:
        return
    var bullet = bullet_scene.instantiate()
    bullet.position = global_position
    # 向玩家方向（需要获取玩家位置）
    var player = get_tree().current_scene.get_node_or_null("Player")
    if player:
        var dir = (player.global_position - global_position).normalized()
        bullet.direction = dir
    else:
        bullet.direction = Vector2.DOWN
    bullet.speed = 200
    bullet.is_player_bullet = false
    bullet.damage = 2
    get_tree().current_scene.add_child(bullet)

func take_damage(amount: int):
    health -= amount
    if health <= 0:
        destroy()

func destroy():
    print("Boss destroyed! Score +", score_value)
    queue_free()
```

> Boss 的分数需要在实例化后通过 `set_score()` 设置，根据当前关卡计算。

### 第六步：玩家与敌机的碰撞（直接撞击）

玩家与敌机碰撞时，双方都应该受到伤害。在 `enemy.gd` 中添加对 `body_entered` 的检测（因为玩家是 `CharacterBody2D`，不是 `Area2D`，所以需要用 `body_entered` 信号）。

修改 `enemy.gd` 的 `_ready()` 和添加新方法：

```gdscript
func _ready():
    add_to_group("enemies")
    bullet_scene = preload("res://scenes/bullet.tscn")
    if can_shoot:
        shoot_timer = randf_range(0.5, 1.5)
    # 连接碰撞信号（注意：需要开启监测）
    body_entered.connect(_on_body_entered)

func _on_body_entered(body):
    if body.name == "Player":
        body.take_damage(1)   # 玩家扣血
        destroy()             # 敌机撞击后也销毁
```

同时，确保玩家节点的 `CollisionShape2D` 所在的根节点是 `CharacterBody2D`，并且玩家脚本有 `take_damage` 方法。

对于 Boss，也需要类似的碰撞处理，但 Boss 撞击后不一定会消失（可以只扣玩家血，Boss 继续存在）。为简化，我们让 Boss 撞击后玩家扣血，Boss 不掉血：

### 第七步：实现全局分数系统（简单单例）

为了在敌机死亡时增加分数，我们需要一个全局可访问的分数管理器。创建自动加载脚本。

```gdscript
extends Node

var score := 0
var current_level := 1

signal score_changed(new_score)

func add_score(amount: int):
    score += amount
    score_changed.emit(score)
    print("Score: ", score)  # 临时打印
```

2. 注册为自动加载：点击菜单 **项目 → 项目设置 → 自动加载**，添加 `global.gd`，节点名称为 `Global`。

现在在任何脚本中可以通过 `Global.add_score(100)` 来增加分数。

修改 `enemy.gd` 的 `destroy()`：

```gdscript
func destroy():

    queue_free()
```

同样，在 `boss.gd` 的 `destroy()` 中：

## 本篇思考题

**1. 在子弹的 `_on_area_entered` 中，我们通过 `other_area.get_parent()` 获取角色节点。这种写法有什么潜在风险？如何改进？**  
> 答案：风险在于，如果角色的碰撞形状不是直接挂在根节点下（例如中间多了一层 `CollisionShape2D` 的父节点不是角色本身），则 `get_parent()` 可能取不到正确的角色节点。改进方法是：给玩家和敌机的根节点设置一个组（如 `player`、`enemy`），然后在子弹中通过 `other_area.get_parent().get_groups()` 检查是否包含对应组，或者使用 `owner` 属性。更可靠的方式是在角色根节点上添加一个自定义属性，并直接获取该节点。

**2. 炸弹调用 `enemy.destroy()` 时，如果敌机正在播放死亡动画或尚未 `queue_free()`，多次调用炸弹会不会重复加分？**  
> 答案：会的。如果炸弹在短时间内多次触发（虽然玩家一般不会连续放炸弹，但理论上可能），同一个敌机会被多次调用 `destroy()`，导致重复加分和多次销毁。解决方案是在 `destroy()` 方法开头增加一个标志 `is_destroyed`，如果已经销毁则直接返回，避免重复执行。

**3. 为什么 Boss 的子弹伤害设置为 2，而普通敌机子弹伤害为 1？这样设计的好处是什么？**  
> 答案：不同敌人造成不同伤害可以增加游戏的挑战性和策略性，让玩家优先躲避 Boss 的攻击。同时，也便于后期设计护甲系统（比如玩家可以通过道具临时减少伤害）。这是一种常见的设计模式。
