### 第二步：为玩家飞机添加基础属性脚本

打开 `player.tscn`，选中根节点 `Player`，修改其脚本 `player.gd`，增加以下属性：

```gdscript
extends CharacterBody2D

# 移动速度
@export var speed := 400.0

# 生命值系统
@export var max_health := 3
var current_health : int

# 武器系统
enum WeaponType { NORMAL, LASER, SHOTGUN, HEAVY }
var current_weapon := WeaponType.NORMAL
# 武器等级（后续升级用，先预留）
var weapon_level := 1

# 射击冷却
@export var shoot_cooldown := 0.2
var shoot_timer := 0.0

# 炸弹系统
@export var bomb_count := 3          # 初始3个炸弹
@export var max_bomb := 9

# 子弹场景引用（稍后设置）
var bullet_scene : PackedScene

# 颜色选项（用于三色选择，在主界面设置后传入）
enum PlayerColor { BLUE, GREEN, RED }
var current_color := PlayerColor.BLUE

# 信号（用于UI更新）
signal health_changed(new_health)
signal bomb_changed(new_count)
signal weapon_changed(weapon_type)

func _ready():
    current_health = max_health
    # 加载子弹场景（注意路径需正确）
    bullet_scene = preload("res://scenes/bullet.tscn")
    # 发射信号初始化UI（如果有的话）
    health_changed.emit(current_health)
    bomb_changed.emit(bomb_count)
    weapon_changed.emit(current_weapon)

func _physics_process(delta):
    # 移动处理（与之前相同）
    var input_dir := Input.get_vector("left", "right", "up", "down")
    velocity = input_dir * speed
    move_and_slide()
    
    var screen_size = get_viewport_rect().size
    position.x = clamp(position.x, 0, screen_size.x)
    position.y = clamp(position.y, 0, screen_size.y)
    
    # 射击冷却计时
    if shoot_timer > 0:
        shoot_timer -= delta
    
    # 射击输入
    if Input.is_action_pressed("shoot") and shoot_timer <= 0:
        shoot()
        shoot_timer = shoot_cooldown

    if Input.is_action_just_pressed("bomb") and bomb_count > 0:

    if Input.is_action_just_pressed("weapon_next"):
        var next = (current_weapon + 1) % WeaponType.size()
        set_weapon(next)
    if Input.is_action_just_pressed("weapon_prev"):
        var prev = (current_weapon - 1 + WeaponType.size()) % WeaponType.size()
        set_weapon(prev)
```

### 第三步：实现射击函数（四种武器的基础版本）

在 `player.gd` 中添加以下函数。注意：这里我们先实现最简单的效果，后续第五篇会增强（升级、伤害不同等）。

```gdscript
func shoot():
    match current_weapon:
        WeaponType.NORMAL:
            # 常规武器：单发子弹，速度中等，伤害1
            fire_bullet(Vector2.UP, 400)
        WeaponType.LASER:
            # 激光：单发，速度更快，伤害2（后面实现伤害）
            fire_bullet(Vector2.UP, 600)
        WeaponType.SHOTGUN:
            # 霰弹：三发散弹，角度偏移
            fire_bullet(Vector2.UP.rotated(-0.2), 400)
            fire_bullet(Vector2.UP, 400)
            fire_bullet(Vector2.UP.rotated(0.2), 400)
        WeaponType.HEAVY:
            # 高威力低速：子弹速度慢，但伤害高（后面实现）
            fire_bullet(Vector2.UP, 200)

func fire_bullet(dir: Vector2, spd: float):
    if bullet_scene == null:
        return
    var bullet = bullet_scene.instantiate()
    bullet.position = global_position  # 从玩家位置发射
    bullet.direction = dir
    bullet.speed = spd
    # 设置子弹类型（用于区分玩家子弹和敌机子弹，以及伤害值）
    bullet.is_player_bullet = true
    # 根据武器类型设置伤害（后续实现伤害系统时使用）
    match current_weapon:
        WeaponType.NORMAL:
            bullet.damage = 1
        WeaponType.LASER:
            bullet.damage = 2
        WeaponType.SHOTGUN:
            bullet.damage = 1
        WeaponType.HEAVY:
            bullet.damage = 3
    get_tree().current_scene.add_child(bullet)

func set_weapon(weapon_type):
    current_weapon = weapon_type
    # 根据武器调整射击冷却（例如激光射速更快，重武器射速慢）
    match current_weapon:
        WeaponType.NORMAL:
            shoot_cooldown = 0.2
        WeaponType.LASER:
            shoot_cooldown = 0.15
        WeaponType.SHOTGUN:
            shoot_cooldown = 0.25
        WeaponType.HEAVY:
            shoot_cooldown = 0.4
    weapon_changed.emit(current_weapon)
```

### 第四步：实现炸弹（AOE）功能

炸弹的效果是：清除屏幕上所有敌机以及敌机子弹，并造成爆炸特效（暂时用简单粒子，后期可美化）。

先添加炸弹逻辑：

```gdscript
func use_bomb():
    if bomb_count <= 0:
        return
    bomb_count -= 1
    bomb_changed.emit(bomb_count)
    
    # 查找场景中所有敌机和敌机子弹并销毁
    var root = get_tree().current_scene
    # 销毁所有敌机（假设敌机场景的根节点组名为 "enemies"）
    for enemy in get_tree().get_nodes_in_group("enemies"):
        enemy.destroy()  # 我们稍后会在敌机脚本中添加 destroy 方法
    # 销毁所有敌机子弹（组名为 "enemy_bullets"）
    for bullet in get_tree().get_nodes_in_group("enemy_bullets"):
        bullet.queue_free()
    
    # 播放炸弹特效（简单起见，可以实例化一个爆炸动画节点）
    # 这里留空，后续补充
    print("Bomb used! Remaining: ", bomb_count)
```

> 注意：为了让炸弹能识别敌机和敌机子弹，我们需要在敌机场景和敌机子弹场景中设置组别。打开 `enemy.tscn`，选中根节点，在检查器最下方的 **Node** 选项卡中，找到 **Groups**，添加组名 `enemies`。同样，敌机子弹（后面制作）添加到组 `enemy_bullets`。

```gdscript
func take_damage(amount: int = 1):

    if current_health <= 0:
        die()
```

## 本篇思考题

**1. 为什么在 `shoot()` 函数中，每次射击都要 `instantiate()` 新的子弹场景？这样做会不会影响性能？**  
> 答案：因为每一颗子弹都是独立的对象，需要单独移动和检测碰撞。Godot 的对象实例化开销并不大，但为了优化，通常可以使用 **对象池（Object Pool）** 来复用子弹。对于小型弹幕游戏，直接实例化完全可以接受。

**2. 炸弹功能中，我们通过 `get_tree().get_nodes_in_group("enemies")` 找到所有敌机并调用 `destroy()`。为什么不在炸弹触发时直接 `queue_free()` 而是调用一个方法？**  
> 答案：因为敌机死亡时可能需要播放爆炸动画、增加玩家分数、产生掉落道具等。如果直接 `queue_free()`，这些逻辑就无法执行。所以统一调用敌机的 `destroy()` 方法，在方法内部处理分数、特效等，最后再 `queue_free()`。

**3. 玩家颜色选择功能中，我们使用 `load()` 来动态加载纹理。如果颜色切换非常频繁，会不会有效率问题？**  
> 答案：通常颜色只在游戏开始时选择一次，不会有性能问题。如果需要在游戏中途多次切换，建议预先加载所有纹理到变量中，切换时直接赋值，避免重复 `load()`。
